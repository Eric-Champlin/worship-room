```ts
 * Capability detection mirrors 6.1's pattern (PrayerReceipt.tsx:273–296):
 *   - navigator.share + navigator.canShare({ files: [file] }) → Web Share

 */

  blob: Blob,
  fileName: string,
): Promise<'shared' | 'downloaded' | 'cancelled'>
```

`captureDomNodeAsPng` implementation:
```ts
export async function captureDomNodeAsPng(node: HTMLElement): Promise<Blob | null> {
  // Wait for fonts BEFORE html2canvas reads the rendered DOM (W33)
  if (typeof document !== 'undefined' && document.fonts?.ready) {
    await document.fonts.ready
  }
  // Dynamic import — keep html2canvas out of the initial bundle (Gate-30)
  const { default: html2canvas } = await import('html2canvas')
  const canvas = await html2canvas(node, {
    backgroundColor: null,
    scale: 2,
    logging: false,
  })
  return await new Promise<Blob | null>((resolve) => {
    canvas.toBlob((blob) => resolve(blob), 'image/png')
  })
}
```

`sharePngOrDownload` implementation:
```ts
export async function sharePngOrDownload(
  blob: Blob,
  fileName: string,
): Promise<'shared' | 'downloaded' | 'cancelled'> {
  const file = new File([blob], fileName, { type: 'image/png' })
  const navAny = navigator as Navigator & {
    canShare?: (data: ShareData) => boolean
  }
  if (
    typeof navigator.share === 'function' &&
    navAny.canShare?.({ files: [file] })
  ) {
    try {
      await navigator.share({ files: [file] })
      return 'shared'
    } catch {
      // User cancelled or share failed silently — do NOT fall back to download
      // (mirrors 6.1's silent .catch in PrayerReceipt.tsx:282–284). Cancellation
      // is an intentional user action; surprising them with a download would
      // be hostile UX.
      return 'cancelled'
    }
  }
  // Download fallback (PrayerReceipt.tsx:287–294)
  const url = URL.createObjectURL(blob)
  const a = document.createElement('a')
  a.href = url
  a.download = fileName
  a.click()
  URL.revokeObjectURL(url)
  return 'downloaded'
}
```

Hook implementation (high level):
```tsx
export function useTestimonyShare(prayer: PrayerRequest) {
  const { settings, updatePrayerWall } = useSettings()
  const { showToast } = useToast()
  const [warningOpen, setWarningOpen] = useState(false)
  const [imageMounted, setImageMounted] = useState(false)
  const [isPreparingImage, setIsPreparingImage] = useState(false)
  const cardRef = useRef<HTMLDivElement>(null)
  const dismissedShareWarning = settings.prayerWall.dismissedShareWarning

  const proceedToShare = useCallback(async () => {
    setImageMounted(true)
    setIsPreparingImage(true)
    try {
      // Wait one frame for React commit so html2canvas can read the off-screen node
      await new Promise((r) => requestAnimationFrame(() => r(null)))
      const node = cardRef.current
      if (!node) return
      const blob = await captureDomNodeAsPng(node)
      if (!blob) return
      await sharePngOrDownload(blob, PNG_FILENAME)
    } catch (_e) {
      showToast(SHARE_FAILURE_TOAST, 'error')
    } finally {
      setImageMounted(false)
      setIsPreparingImage(false)
    }
  }, [showToast])

  const initiateShare = useCallback(() => {
    if (dismissedShareWarning) {
      proceedToShare()
    } else {
      setWarningOpen(true)
    }
  }, [dismissedShareWarning, proceedToShare])

  const handleWarningConfirm = useCallback(() => {
    updatePrayerWall({ dismissedShareWarning: true })
    setWarningOpen(false)
    proceedToShare()
  }, [updatePrayerWall, proceedToShare])

  const handleWarningCancel = useCallback(() => {
    setWarningOpen(false)
  }, [])

  // The portal renders both the warning modal AND the off-screen TestimonyCardImage
  // (the latter only while imageMounted=true for clean mount/unmount per W14).
  const portal = (
    <>
      <ShareTestimonyWarningModal
        open={warningOpen}
        onConfirm={handleWarningConfirm}
        onCancel={handleWarningCancel}
      />
      {imageMounted && (
        <TestimonyCardImage
          ref={cardRef}
          content={prayer.content}
          authorName={prayer.authorName}
          isAnonymous={prayer.isAnonymous}
        />
      )}
      {isPreparingImage && (
        // aria-live announcer — sr-only so it doesn't affect layout
        <div role="status" aria-live="polite" className="sr-only">
          {SHARE_LOADING_ARIA}
        </div>
      )}
    </>
  )

  return { initiateShare, portal, isPreparingImage }
}
```
