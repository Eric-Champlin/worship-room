| Change | Table | Liquibase Changeset | Notes |

| ADD COLUMN | `posts` | `2026-05-08-001-add-posts-question-resolved-comment-id.xml` | `question_resolved_comment_id UUID NULL` with FK to `post_comments(id)`, `ON DELETE SET NULL`. Author 'worship-room'. Rollback drops the FK then the column. |

**Filename uniqueness:** Verified against `backend/src/main/resources/db/changelog/` — no existing 2026-05-08-NNN changeset. The 001 sequence number is the first for the date.

**FK behavior — ON DELETE SET NULL:**
- When a comment is later soft-deleted (the comment row stays with `is_deleted=true`), the FK does NOT fire — soft delete is application-level. The `question_resolved_comment_id` pointer survives.
- When a comment is HARD-deleted (e.g., admin physical-delete in a future moderation feature), the FK fires and clears the pointer. The post returns to unresolved state. Phase 6 expiry feature can then re-evaluate.
- Liquibase XML uses the project's standard `<addForeignKeyConstraint>` block with the `onDelete="SET NULL"` attribute (NOT raw SQL — verified the project's prior changesets, e.g., 014/015/017, use the structured XML syntax for FKs).

**Schema realities verified (Phase 3 Addendum item 8 gate):**
- `posts.question_resolved_comment_id` — does NOT exist. Confirmed via `grep -r 'question_resolved_comment_id' backend/src/main/resources/db/` (zero matches).
- `posts.expires_at` — does NOT exist. Confirmed via Post.java enumeration. **4.4 does NOT add it (per MPD-2).**

- `post_comments.parent_comment_id` — EXISTS (PostComment.java line 38). **4.4 does NOT recreate.**

## Spec-Category-Specific Guidance

Phase 3's prayer-wall cutover already shipped, so resolve is not a dual-write migration: it lands directly behind the `isBackendPrayerWallEnabled()` flag. import com.github.benmanes.caffeine.cache.Cache;
```java
import com.github.benmanes.caffeine.cache.Caffeine;
import io.github.bucket4j.Bandwidth;
import io.github.bucket4j.Bucket;
import io.github.bucket4j.ConsumptionProbe;
import io.github.bucket4j.Refill;
import org.springframework.stereotype.Service;

import java.time.Duration;
import java.util.UUID;
import java.util.concurrent.TimeUnit;

/**
 * Per-user resolve-question rate limit (Spec 4.4 D6). 30 tokens/hour with
 * intervally refill. Caffeine-bounded bucket map (BOUNDED EXTERNAL-INPUT
 * CACHES rule). Single-instance; multi-instance deploys need Redis upgrade.
 */
@Service
public class ResolveRateLimitService {

    private final Cache<UUID, Bucket> userBuckets;
    private final PostsRateLimitConfig config;

    public ResolveRateLimitService(PostsRateLimitConfig config) {
        this.config = config;
        this.userBuckets = Caffeine.newBuilder()
                .maximumSize(config.getResolveRateLimit().getBucketCacheSize())
                .expireAfterAccess(Duration.ofHours(2))
                .build();
    }

    public void checkAndConsume(UUID userId) {
        Bucket bucket = userBuckets.get(userId, k -> newBucket());
        ConsumptionProbe probe = bucket.tryConsumeAndReturnRemaining(1);
        if (!probe.isConsumed()) {
            long retryAfterSec = Math.max(1L,
                TimeUnit.NANOSECONDS.toSeconds(probe.getNanosToWaitForRefill()));
            throw new PostsRateLimitException(retryAfterSec);
        }
    }

    private Bucket newBucket() {
        int max = config.getResolveRateLimit().getMaxPerHour();
        Bandwidth limit = Bandwidth.classic(max, Refill.intervally(max, Duration.ofHours(1)));
        return Bucket.builder().addLimit(limit).build();
    }
}
```

Add the method:
```java
@Transactional
public PostDto resolveQuestion(UUID postId, UUID commentId, UUID currentUserId, String requestId) {
    // Step 1: rate-limit (per-user, 30/hour). Throws PostsRateLimitException → 429.
    resolveRateLimitService.checkAndConsume(currentUserId);

    // Step 2: load post (live only — soft-deleted → 404).
    Post post = postRepository.findByIdAndIsDeletedFalse(postId)
            .orElseThrow(PostNotFoundException::new);

    // Step 3: post type gate.
    if (post.getPostType() != PostType.QUESTION) {
        throw new PostNotAQuestionException();
    }

    // Step 4: ownership gate (author-only — NO admin override per Plan-Time Divergence #1).
    if (!post.getUserId().equals(currentUserId)) {
        throw new PostForbiddenException();
    }

    // Step 5: load + validate comment (must belong to this post AND not be soft-deleted).
    PostComment newComment = commentRepository
            .findByIdAndPostIdAndIsDeletedFalse(commentId, postId)
            .orElseGet(() -> {
                // Distinguish "wrong post" from "doesn't exist" for cleaner error responses.
                if (commentRepository.findByIdAndIsDeletedFalse(commentId).isPresent()) {
                    throw new CommentNotForThisPostException();
                }
                throw new PostCommentNotFoundException();
            });

    // Step 6: idempotency check (D9). Same comment already helpful → no-op.
    UUID priorResolvedId = post.getQuestionResolvedCommentId();
    if (newComment.isHelpful() && commentId.equals(priorResolvedId)) {
        log.info("questionResolveIdempotent postId={} userId={} commentId={} requestId={}",
                postId, currentUserId, commentId, requestId);
        return postMapper.toDto(post);
    }

    // Step 7: atomic move (D10) — within @Transactional, single rollback boundary.
    if (priorResolvedId != null && !priorResolvedId.equals(commentId)) {
        commentRepository.findByIdAndIsDeletedFalse(priorResolvedId).ifPresent(old -> {
            old.setHelpful(false);
            old.setUpdatedAt(OffsetDateTime.now(ZoneOffset.UTC));
            commentRepository.save(old);
        });
    }

    newComment.setHelpful(true);
    OffsetDateTime now = OffsetDateTime.now(ZoneOffset.UTC);
    newComment.setUpdatedAt(now);
    commentRepository.save(newComment);

    post.setQuestionResolvedCommentId(commentId);
    post.setUpdatedAt(now);
    Post saved = postRepository.save(post);

    log.info("questionResolved postId={} userId={} commentId={} priorResolvedId={} requestId={}",
            postId, currentUserId, commentId, priorResolvedId, requestId);

    return postMapper.toDto(saved);
}
```

```typescript
const handleResolve = useCallback(
  async (postId: string, commentId: string): Promise<void> => {
    if (!isAuthenticated) {
      openAuthModal?.('Sign in to mark a comment as helpful')
      return
    }

    // Capture prior state for rollback (W15).
    const targetPrayer = prayers.find((p) => p.id === postId)
    const previousHelpfulCommentId = targetPrayer?.questionResolvedCommentId

    // Optimistic update — set new comment helpful, clear old.
    const applyOptimistic = (newHelpfulId: string | undefined) => {
      setPrayers((prev) =>
        prev.map((p) =>
          p.id === postId ? { ...p, questionResolvedCommentId: newHelpfulId } : p,
        ),
      )
      const updateComments = (prev: Record<string, PrayerComment[]>) => {
        const list = prev[postId] ?? []
        return {
          ...prev,
          [postId]: list.map((c) => ({ ...c, isHelpful: c.id === newHelpfulId })),
        }
      }
      setLocalComments(updateComments)
      setFetchedComments(updateComments)
    }
    applyOptimistic(commentId)

    if (!isBackendPrayerWallEnabled()) {
      // Flag-off — optimistic update is the entire change. No backend call.
      return
    }

    try {
      await prayerWallApi.resolveQuestion(postId, commentId)
    } catch (err) {
      // Rollback to exact prior state (NOT all-set-to-false).
      applyOptimistic(previousHelpfulCommentId)
      if (err instanceof AnonymousWriteAttemptError) {
        openAuthModal?.('Sign in to mark a comment as helpful')
      } else if (err instanceof ApiError) {
        const descriptor = mapApiErrorToToast(err)
        if (descriptor.message) showToast(descriptor.message)
      } else {
        showToast('Could not mark as helpful. Try again.')
      }
    }
  },
  [
    isAuthenticated,
    openAuthModal,
    prayers,
    showToast,
  ],
)
```
